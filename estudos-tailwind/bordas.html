Border-0 = borda, valor refere ao width da borda.

border-x-o = bordas horizontais - left/right

border-y-o = bordas verticais - top/bottom


border-l-o = border-left-width.

border-r-o = border-right-width.

border-t-o = border-top-width.

border-b-o = border-bottom-width.

border-radius -> borda arredondada.

rounded-none = border-radius none.

rounded = border-radius normal.


rounded-sm = border-radius small.

rounded-md = border-radius medium.

rounded-lg = border-radius large.

rounded-xlg = border-radius extra large.
