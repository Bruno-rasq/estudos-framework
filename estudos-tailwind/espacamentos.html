w = width

h = height


p = padding, o valor é referente ao tamanho do padding e para todos os lados.

px = padding horizontal, esquerda e direita.

py = padding vertical, top e bottom

pl = padding esquerda.

pr = padding direita.

pt = padding top.

pb = padding bottom.


m = margin, o valor é referente ao tamanho do margin e para todos os lados.

mx = margin horizontal, esquerda e direita.

my = margin vertical, top e bottom

ml = margin esquerda.

mr = margin direita.

mt = margin top.

mb = margin bottom.

space-x/y = space between.

space-x = space between quando flex-direction for row.

space-y = space between quando flex-direction for column.
